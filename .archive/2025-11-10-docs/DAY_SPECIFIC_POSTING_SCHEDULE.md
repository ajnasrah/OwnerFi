  sunday: [6, 12, 8, 10],           // 6 AM, 12 PM, 8 AM, 10 AM

  thursday: [9, 11, 8, 10, 16],     // 9 AM, 11 AM, 8 AM, 10 AM, 4 PM
  friday: [17, 14, 19, 11, 8],      // 5 PM, 2 PM, 7 PM, 11 AM, 8 AM
  saturday: [21, 12, 13, 10, 18],   // 9 PM, 12 PM, 1 PM, 10 AM, 6 PM
};

// Get posting times for today
export function getTodaysPostingTimes(): number[] {
  const days = ["sunday", "monday", "tuesday", "wednesday", "thursday", "friday", "saturday"];
  const today = new Date().getDay();
  return DAY_SPECIFIC_POSTING_TIMES[days[today]] || [];
}
